# L3 Reviewer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a `/l3-reviewer/` subpage that accepts a Word transcript of a NZ Police Level 3 investigative interview, reviews it with Claude against the PEACE model and Level 3 moderation form, and returns a structured on-screen report plus downloadable Word and Markdown files.

**Architecture:** A wizard UI (3 input steps → progress → results) posts a multipart form to `POST /api/l3-reviewer/upload`. The server extracts the transcript with mammoth, calls Claude for a structured JSON assessment, generates both report formats in memory, and streams progress via SSE. Results and download tokens are returned in the `done` event.

**Tech Stack:** Node.js/Express, mammoth (DOCX extraction), @anthropic-ai/sdk (Claude Sonnet 4.6), pizzip (DOCX generation), SSE, vanilla HTML/CSS/JS frontend, Node built-in test runner.

    `Detailed knowledge of investigation: ${formData.detailedKnowledge || 'Not provided'}`,
    `Planning comments: ${formData.planningComments || 'Not provided'}`,
    '',
    '## Interview Transcript',
    transcriptText,
  ];
  return lines.join('\n');

/**
 * Review an interview transcript against the L3 moderation form.
 *
 * @param {string}   transcriptText - Plain text from DOCX
 * @param {Object}   formData       - All wizard form fields
 * @param {Function} [onProgress]   - Optional callback({ type: 'connected' })
 * @returns {Promise<Object>}        Parsed JSON assessment
 */
export async function reviewInterview(transcriptText, formData, onProgress) {
  const apiKey = process.env.CLAUDE_API_KEY || process.env.ANTHROPIC_API_KEY;
  if (!apiKey) throw new Error('CLAUDE_API_KEY (or ANTHROPIC_API_KEY) is not set');
  const client = new Anthropic({ apiKey });

  const userContent = buildReviewPrompt(transcriptText, formData);

  let raw = '';
  let connected = false;

  const stream = client.messages.stream({
    model: 'claude-sonnet-4-6',
    max_tokens: 8000,
    system: SYSTEM_PROMPT,
    messages: [{ role: 'user', content: userContent }],
  });

  stream.on('text', (text) => {
    raw += text;
    if (!connected) {
      connected = true;
      onProgress?.({ type: 'connected' });
    }
  });

  const final = await stream.finalMessage();

  if (final.stop_reason === 'max_tokens') {
    throw new Error('The AI review was cut off before completing. Try a shorter transcript or contact support.');
  }

  const cleaned = raw.trim().replace(/^```(?:json)?\s*/i, '').replace(/\s*```$/, '');

  let result;
  try {
    result = JSON.parse(cleaned);
  } catch {
    const likelyTruncated = !cleaned.trimEnd().endsWith('}');
    console.error(`[interviewReviewer] JSON parse failed. Length: ${cleaned.length}. Last 100: ${cleaned.slice(-100)}`);
    throw new Error(
      likelyTruncated
        ? 'The AI review was cut off before completing. Try a shorter transcript or contact support.'
        : 'Claude returned invalid JSON. Check server logs for details.'
    );
  }

  if (!result.verdict || !result.section5 || !result.section6 || !result.section7 || !result.section8) {
    throw new Error('Claude response is missing required assessment sections.');
  }

  return result;
}

/**
 * Escape XML special characters.
 * @param {string} str
 * @returns {string}
 */
function xmlEsc(str) {
  return String(str || '')
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');
}

/**
 * Build a Word paragraph XML element.
 * @param {string}  text
 * @param {Object}  [opts]
 * @param {boolean} [opts.bold]
 * @param {string}  [opts.size]   - half-point size, e.g. "28" = 14pt
 * @param {string}  [opts.color]  - hex color without #, e.g. "22C55E"
 * @returns {string}
 */
function para(text, opts = {}) {
  const rpr = [
    opts.bold ? '<w:b/>' : '',
    opts.size ? `<w:sz w:val="${opts.size}"/><w:szCs w:val="${opts.size}"/>` : '',
    opts.color ? `<w:color w:val="${opts.color}"/>` : '',
  ].join('');

  return (
    `<w:p>` +
    `<w:r>` +
    (rpr ? `<w:rPr>${rpr}</w:rPr>` : '') +
    `<w:t xml:space="preserve">${xmlEsc(text)}</w:t>` +
    `</w:r>` +
    `</w:p>`
  );
}

/**
 * Build a two-column table row.
 * @param {string} label
 * @param {string} value
 * @returns {string}
 */
function tableRow(label, value) {
  const cell = (text, bold = false) =>
    `<w:tc>` +
    `<w:tcPr><w:tcBorders>` +
    `<w:top w:val="single" w:sz="4" w:color="404040"/>` +
    `<w:bottom w:val="single" w:sz="4" w:color="404040"/>` +
    `<w:left w:val="single" w:sz="4" w:color="404040"/>` +
    `<w:right w:val="single" w:sz="4" w:color="404040"/>` +
    `</w:tcBorders></w:tcPr>` +
    `<w:p><w:r>` +
    (bold ? '<w:rPr><w:b/></w:rPr>' : '') +
    `<w:t xml:space="preserve">${xmlEsc(text)}</w:t>` +
    `</w:r></w:p>` +
    `</w:tc>`;

  return `<w:tr>${cell(label, true)}${cell(value)}</w:tr>`;
}

/**
 * Build a three-column checklist table row (item | result | comment).
 * @param {Object} item
 * @returns {string}
 */
function checklistRow(item) {
  const result = item.result || item.frequency || '';
  const comment = item.comment || '';
  const cell = (text) =>
    `<w:tc>` +
    `<w:tcPr><w:tcBorders>` +
    `<w:top w:val="single" w:sz="4" w:color="CCCCCC"/>` +
    `<w:bottom w:val="single" w:sz="4" w:color="CCCCCC"/>` +
    `<w:left w:val="single" w:sz="4" w:color="CCCCCC"/>` +
    `<w:right w:val="single" w:sz="4" w:color="CCCCCC"/>` +
    `</w:tcBorders></w:tcPr>` +
    `<w:p><w:r><w:t xml:space="preserve">${xmlEsc(text)}</w:t></w:r></w:p>` +
    `</w:tc>`;
  return `<w:tr>${cell(item.item)}${cell(result)}${cell(comment)}</w:tr>`;
}

/**
 * Build a three-column checklist header row.

 * @returns {string}

function checklistHeader(col2Header) {
  const cell = (text) =>
    `<w:tc>` +
    `<w:tcPr><w:shd w:val="clear" w:color="auto" w:fill="E2E8F0"/><w:tcBorders>` +

// server/routes/l3Reviewer.js

import { Router } from 'express';
import multer from 'multer';
import { mkdirSync } from 'fs';
import { existsSync, unlinkSync } from 'fs';
import { randomUUID } from 'crypto';
import mammoth from 'mammoth';
import { reviewInterview } from '../lib/interviewReviewer.js';
import { buildMarkdownReport, buildDocxBuffer } from '../lib/l3ReportGenerator.js';

const UPLOAD_DIR = '/tmp/l3_uploads';
mkdirSync(UPLOAD_DIR, { recursive: true });

const upload = multer({
  dest: UPLOAD_DIR,
  limits: { fileSize: 50 * 1024 * 1024 }, // 50 MB
  fileFilter: (_req, file, cb) => {
    const ok = /\.docx$/i.test(file.originalname);
    cb(ok ? null : new Error('Only .docx files are supported'), ok);
  },
});

// In-memory job store — stores report buffers, never writes to disk
const jobs = new Map();
const JOB_TTL_MS = 15 * 60 * 1000; // 15 min

export const l3ReviewerRouter = Router();

// ── Helpers ────────────────────────────────────────────────────────────────────

function pushEvent(jobId, data) {
  const job = jobs.get(jobId);
  if (!job) return;
  job.events.push(data);
  job.listeners.forEach(send => send(data));
}

function finishJob(jobId, result, error) {
  const job = jobs.get(jobId);
  if (!job) return;
  job.done   = true;
  job.result = result;
  job.error  = error || null;
  job.listeners.forEach(send => send({ step: error ? 'error' : 'done', result, error }));
  job.listeners = [];
  job.cleanupTimer = setTimeout(() => jobs.delete(jobId), JOB_TTL_MS);
}

function safeUnlink(path) {
  try { if (path && existsSync(path)) unlinkSync(path); } catch {}
}

// ── POST /api/l3-reviewer/upload ───────────────────────────────────────────────

l3ReviewerRouter.post('/upload', (req, res, next) => {
  upload.single('transcript')(req, res, (err) => {
    if (err) {
      if (err.code === 'LIMIT_FILE_SIZE') {
        return res.status(413).json({ error: 'File exceeds the 50 MB limit.' });
      }
      return res.status(400).json({ error: err.message || 'Upload failed' });
    }
    next();
  });
}, (req, res) => {
  if (!req.file) return res.status(400).json({ error: 'No transcript file uploaded.' });

  // Parse form fields — multiselect fields are sent as comma-separated strings or arrays
  const parseMulti = (v) => {
    if (!v) return [];
    if (Array.isArray(v)) return v;
    return v.split(',').map(s => s.trim()).filter(Boolean);
  };

  const formData = {
    dateOfInterview:        req.body.dateOfInterview        || '',
    reasonForInterview:     req.body.reasonForInterview     || '',
    fileNumber:             req.body.fileNumber             || '',
    lengthMinutes:          req.body.lengthMinutes          || '',
    interviewerName:        req.body.interviewerName        || '',
    interviewerQid:         req.body.interviewerQid         || '',
    interviewerSection:     req.body.interviewerSection     || '',
    interviewerSupervisor:  req.body.interviewerSupervisor  || '',
    wellcheckAcknowledged:  req.body.wellcheckAcknowledged  || '',
    firstTimeAccreditation: req.body.firstTimeAccreditation || '',
    assessorName:           req.body.assessorName           || '',
    assessorQid:            req.body.assessorQid            || '',
    dateEvaluated:          req.body.dateEvaluated          || '',
    dateFeedbackGiven:      req.body.dateFeedbackGiven      || '',
    intervieweeName:        req.body.intervieweeName        || '',
    intervieweeGender:      req.body.intervieweeGender      || '',
    specialConsiderations:  parseMulti(req.body.specialConsiderations),
    otherPersonsPresent:    parseMulti(req.body.otherPersonsPresent),
    supportingDocuments:    parseMulti(req.body.supportingDocuments),
    planningNotes:          req.body.planningNotes          || '',
    detailedKnowledge:      req.body.detailedKnowledge      || '',
    planningComments:       req.body.planningComments       || '',
    enquiriesIdentified:    req.body.enquiriesIdentified    || '',
    whatWentWell:           req.body.whatWentWell           || '',
    learningPoints:         req.body.learningPoints         || '',
    assessorPositiveFeedback: req.body.assessorPositiveFeedback || '',
    assessorLearningPoints: req.body.assessorLearningPoints || '',
    learningDevelopmentPlan: req.body.learningDevelopmentPlan || '',
  };

  const jobId = randomUUID();
  jobs.set(jobId, {
    events:      [],
    listeners:   [],
    result:      null,
    done:        false,
    error:       null,
    mdReport:    null,   // Markdown string
    docxBuffer:  null,   // DOCX Buffer
    cleanupTimer: null,
  });

  runPipeline(jobId, req.file, formData).catch(err => {
    console.error('[l3Reviewer] pipeline error:', err);
    pushEvent(jobId, { step: 'error', message: err.message });
    finishJob(jobId, null, err.message);
  });

  res.json({ jobId });
});

// ── GET /api/l3-reviewer/status/:jobId (SSE) ──────────────────────────────────

l3ReviewerRouter.get('/status/:jobId', (req, res) => {
  const job = jobs.get(req.params.jobId);
  if (!job) return res.status(404).json({ error: 'Job not found' });

  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.flushHeaders();

  const send = (data) => res.write(`data: ${JSON.stringify(data)}\n\n`);

  // Replay buffered events for reconnecting clients
  job.events.forEach(send);

  if (job.done) {
    send({ step: job.error ? 'error' : 'done', result: job.result, error: job.error });
    return res.end();
  }

  job.listeners.push(send);
  req.on('close', () => {
    job.listeners = job.listeners.filter(l => l !== send);
  });
});

// ── GET /api/l3-reviewer/download/:jobId/docx ─────────────────────────────────

l3ReviewerRouter.get('/download/:jobId/docx', (req, res) => {
  const job = jobs.get(req.params.jobId);
  if (!job || !job.done) return res.status(404).json({ error: 'Job not ready or not found' });
  if (job.error) return res.status(500).json({ error: job.error });
  if (!job.docxBuffer) return res.status(404).json({ error: 'DOCX report not found' });

  res.setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document');
  res.setHeader('Content-Disposition', 'attachment; filename="l3-interview-review.docx"');
  res.send(job.docxBuffer);
});

// ── GET /api/l3-reviewer/download/:jobId/md ───────────────────────────────────

l3ReviewerRouter.get('/download/:jobId/md', (req, res) => {
  const job = jobs.get(req.params.jobId);
  if (!job || !job.done) return res.status(404).json({ error: 'Job not ready or not found' });
  if (job.error) return res.status(500).json({ error: job.error });
  if (!job.mdReport) return res.status(404).json({ error: 'Markdown report not found' });

  res.setHeader('Content-Type', 'text/markdown; charset=utf-8');
  res.setHeader('Content-Disposition', 'attachment; filename="l3-interview-review.md"');
  res.send(job.mdReport);
});

// ── Pipeline ───────────────────────────────────────────────────────────────────

async function runPipeline(jobId, transcriptFile, formData) {
  const job = jobs.get(jobId);

  try {
    // Step 1: Extract transcript text
    pushEvent(jobId, { step: 'extracting', message: 'Extracting text from transcript…' });

    let transcriptText;
    try {
      const { value } = await mammoth.extractRawText({ path: transcriptFile.path });
      transcriptText = value;
    } catch (err) {
      throw new Error('Could not read this file — please ensure it is a valid Word document (.docx).');
    }

    const wordCount = transcriptText.trim().split(/\s+/).length;
    pushEvent(jobId, {
      step: 'extracting_done',
      message: `${wordCount.toLocaleString()} words extracted`,
      wordCount,
    });

    // Step 2: AI review
    pushEvent(jobId, { step: 'reviewing', message: 'Sending transcript to Claude Sonnet 4.6…' });

    let review;
    const heartbeat = setInterval(() => pushEvent(jobId, { step: 'heartbeat' }), 30_000);
    try {
      review = await reviewInterview(transcriptText, formData, ({ type }) => {
        if (type === 'connected') {
          pushEvent(jobId, { step: 'reviewing_connected', message: 'Claude is generating the assessment…' });
        }
      });
    } finally {
      clearInterval(heartbeat);
    }

    pushEvent(jobId, { step: 'reviewing_done', message: `Assessment complete — ${review.verdict}` });

    // Step 3: Generate reports
    pushEvent(jobId, { step: 'generating', message: 'Generating Word and Markdown reports…' });

    job.mdReport   = buildMarkdownReport(formData, review);
    job.docxBuffer = buildDocxBuffer(formData, review);

    pushEvent(jobId, { step: 'generating_done', message: 'Reports ready' });

    // Done
    finishJob(jobId, {
      verdict:         review.verdict,
      narrativeSummary: review.narrativeSummary,
      section5:        review.section5,
      section6:        review.section6,
      section7:        review.section7,
      section8:        review.section8,
      strengths:       review.strengths,
      learningPoints:  review.learningPoints,
      aiSuggestedFeedback: review.aiSuggestedFeedback,
    });

  } finally {
    safeUnlink(transcriptFile.path);
  }
}

    // ── State ────────────────────────────────────────────────────────────────────
    let currentJobId = null;
    let reviewerTimer = null;
    let reviewerStart = null;

    // Collected form data across steps
    const formState = {};

    // ── Navigation ────────────────────────────────────────────────────────────────

    function showScreen(id) {
      document.querySelectorAll('.l3-screen').forEach(s => s.hidden = true);
      document.getElementById(id).hidden = false;
    }

    // ── Toggle buttons (Yes/No) ───────────────────────────────────────────────────

    document.querySelectorAll('.toggle-group').forEach(group => {
      group.querySelectorAll('.toggle-btn').forEach(btn => {
        btn.addEventListener('click', () => {
          group.querySelectorAll('.toggle-btn').forEach(b => b.classList.remove('selected'));
          btn.classList.add('selected');
        });
      });
    });

    // ── Multi-select chips ────────────────────────────────────────────────────────

    document.querySelectorAll('.multi-select').forEach(ms => {
      ms.querySelectorAll('.multi-chip').forEach(chip => {
        chip.addEventListener('click', () => {
          chip.classList.toggle('selected');
        });
      });

    // ── Collect form data helpers ─────────────────────────────────────────────────

    function collectInputs(container) {
      const data = {};
      container.querySelectorAll('input[name], select[name], textarea[name]').forEach(el => {
        if (el.type === 'file') return;
        data[el.name] = el.value;
      });
      container.querySelectorAll('.toggle-group[data-name]').forEach(group => {
        const selected = group.querySelector('.toggle-btn.selected');
        data[group.dataset.name] = selected ? selected.dataset.value : '';
      });
      container.querySelectorAll('.multi-select[data-name]').forEach(ms => {
        data[ms.dataset.name] = Array.from(ms.querySelectorAll('.multi-chip.selected')).map(c => c.dataset.value);
      });
      return data;
    }

    function showAdminError(msg) {
      const panel = document.getElementById('admin-error');
      panel.querySelector('p').textContent = msg;
      panel.classList.add('visible');
    }

    function hideAdminError() {
      document.getElementById('admin-error').classList.remove('visible');
    }

    // ── Step 1 → Step 2 ──────────────────────────────────────────────────────────

    document.getElementById('admin-form').addEventListener('submit', (e) => {
      e.preventDefault();
      hideAdminError();
      const file = document.getElementById('transcript-input').files[0];
      if (!file) return showAdminError('Please select a .docx transcript file.');
      if (!/\.docx$/i.test(file.name)) return showAdminError('Only .docx files are supported.');
      Object.assign(formState, collectInputs(document.getElementById('step-admin')));
      formState._file = file;
      showScreen('step-planning');
    });

    // ── Step 2 back/next ─────────────────────────────────────────────────────────

    document.getElementById('back-to-planning').addEventListener('click', () => showScreen('step-planning'));
    document.getElementById('submit-review').addEventListener('click', () => {
      Object.assign(formState, collectInputs(document.getElementById('step-reflection')));
      showScreen('step-progress');
      startReview();
    });

    // ── Progress helpers ─────────────────────────────────────────────────────────

    function setStepState(key, state, note) {
      const icon = document.getElementById(`si-${key}`);
      const noteEl = document.getElementById(`sn-${key}`);
      icon.className = `step-icon ${state}`;
      if (state === 'active') icon.innerHTML = '<span class="spinner">⟳</span>';
      else if (state === 'done') icon.innerHTML = '✓';
      else if (state === 'error') icon.innerHTML = '✗';
      if (note !== undefined && noteEl) noteEl.textContent = note;
    }

    function showProgressError(msg) {
      const panel = document.getElementById('progress-error');
      document.getElementById('progress-error-msg').textContent = msg;
      panel.classList.add('visible');
      setStepState('done', 'error');
    }

    function elapsedStr(startMs) {
      const s = Math.floor((Date.now() - startMs) / 1000);
      return s < 60 ? `${s}s` : `${Math.floor(s/60)}m ${s%60}s`;
    }

    // ── Submit review ────────────────────────────────────────────────────────────

    async function startReview() {
      setStepState('uploading', 'active', 'Sending transcript…');

      const fd = new FormData();
      fd.append('transcript', formState._file);

      // Append all scalar form fields
      const skip = new Set(['_file']);
      for (const [k, v] of Object.entries(formState)) {
        if (skip.has(k)) continue;
        if (Array.isArray(v)) {
          fd.append(k, v.join(','));
        } else {
          fd.append(k, v);
        }
      }

      let jobId;
      try {
        const r = await fetch('/api/l3-reviewer/upload', { method: 'POST', body: fd });
        const json = await r.json();
        if (!r.ok) throw new Error(json.error || 'Upload failed');
        jobId = json.jobId;
        currentJobId = jobId;
      } catch (err) {
        return showProgressError(err.message);
      }

      setStepState('uploading', 'done', 'Uploaded');
      setStepState('extracting', 'active');

      // Open SSE connection
      const es = new EventSource(`/api/l3-reviewer/status/${jobId}`);

      es.onmessage = (e) => {
        const data = JSON.parse(e.data);
        handleSseEvent(data, jobId);
        if (data.step === 'done' || data.step === 'error') es.close();
      };

      es.onerror = () => {
        es.close();
        showProgressError('Connection lost. Please refresh and try again.');
      };
    }

    function handleSseEvent(data, jobId) {
      switch (data.step) {
        case 'extracting':
          setStepState('extracting', 'active', data.message);
          break;
        case 'extracting_done':
          setStepState('extracting', 'done', data.message);
          setStepState('reviewing', 'active', 'Connecting to Claude…');
          reviewerStart = Date.now();
          reviewerTimer = setInterval(() => {
            setStepState('reviewing', 'active', `Generating assessment… ${elapsedStr(reviewerStart)}`);
          }, 1000);
          break;
        case 'reviewing_connected':
          setStepState('reviewing', 'active', `Generating assessment… ${elapsedStr(reviewerStart)}`);
          break;
        case 'reviewing_done':
          clearInterval(reviewerTimer);
          setStepState('reviewing', 'done', data.message);
          setStepState('generating', 'active', 'Building reports…');
          break;
        case 'generating_done':
          setStepState('generating', 'done', data.message);
          break;
        case 'heartbeat':
          // Keep-alive — no UI change needed
          break;
        case 'done':
          setStepState('done', 'done', 'Assessment complete');
          setTimeout(() => renderResults(data.result, jobId), 600);
          break;
        case 'error':
          clearInterval(reviewerTimer);
          showProgressError(data.error || data.message || 'An error occurred.');
          break;
      }
    }

    // ── Render results ────────────────────────────────────────────────────────────

    function resultClass(val) {
      if (!val) return '';
      const v = val.toLowerCase();
      if (v === 'yes') return 'result-yes';
      if (v === 'no') return 'result-no';
      if (v === 'n/a') return 'result-na';
      if (v === 'often') return 'result-often';
      if (v === 'sometimes') return 'result-sometimes';
      if (v === 'hardly ever') return 'result-hardly';
      if (v === 'never') return 'result-never';
      return '';
    }

    function buildChecklistTable(tableId, items, valueKey) {
      const tbody = document.querySelector(`#${tableId} tbody`);
      tbody.innerHTML = '';
      items.forEach(item => {
        const val = item[valueKey] || item.result || item.frequency || '';
        const tr = document.createElement('tr');
        tr.innerHTML =
          `<td>${item.item}</td>` +
          `<td><span class="${resultClass(val)}">${val}</span></td>` +
          `<td style="color:#64748b;font-size:12px;">${item.comment || ''}</td>`;
        tbody.appendChild(tr);
      });
    }

    function setRatingBar(barId, scoreId, badgeId, rating, label) {
      document.getElementById(barId).style.width = `${(rating / 5) * 100}%`;
      document.getElementById(scoreId).textContent = `${rating}/5 — ${label}`;
      if (badgeId) document.getElementById(badgeId).textContent = `${rating}/5 ${label}`;
    }

    function renderResults(result, jobId) {
      // Verdict
      const isCompetent = result.verdict === 'COMPETENT';
      const banner = document.getElementById('verdict-banner');
      banner.className = `verdict-banner ${isCompetent ? 'competent' : 'not-competent'}`;
      const vt = document.getElementById('verdict-text');
      vt.textContent = result.verdict;
      vt.className = `verdict-text ${isCompetent ? 'competent' : 'not-competent'}`;
      document.getElementById('verdict-summary').textContent = result.narrativeSummary;

      // Section rating bars
      setRatingBar('bar-s5', 'score-s5', 'badge-s5', result.section5.rating, result.section5.ratingLabel);
      setRatingBar('bar-s6', 'score-s6', 'badge-s6', result.section6.rating, result.section6.ratingLabel);
      setRatingBar('bar-s7', 'score-s7', 'badge-s7', result.section7.rating, result.section7.ratingLabel);

      // Strengths and learning points
      const sl = document.getElementById('strengths-list');
      sl.innerHTML = result.strengths.map(s => `<li>${s}</li>`).join('');
      const ll = document.getElementById('learning-list');
      ll.innerHTML = result.learningPoints.map(l => `<li>${l}</li>`).join('');

      // Checklist tables
      buildChecklistTable('table-s5', result.section5.checklist, 'result');
      buildChecklistTable('table-s6', result.section6.checklist, 'result');
      buildChecklistTable('table-s7', result.section7.items, 'frequency');
      buildChecklistTable('table-s8', result.section8.checklist, 'result');

      // Download buttons
      document.getElementById('dl-docx').onclick = () => {
        window.location.href = `/api/l3-reviewer/download/${jobId}/docx`;
      };
      document.getElementById('dl-md').onclick = () => {
        window.location.href = `/api/l3-reviewer/download/${jobId}/md`;
      };

      showScreen('step-results');
    }
